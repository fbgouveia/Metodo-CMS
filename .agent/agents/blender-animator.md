```python
def animar_com_driver(objeto, propriedade, expressao_python):
    """
    Cria animação procedural via Driver (equação matemática).
    Mais eficiente que keyframes manuais para movimentos repetitivos.
    
    Exemplo: objeto.location.z = sin(#frame / 10) * 2
    """
    objeto.driver_add(propriedade)
    driver = objeto.animation_data.drivers[-1].driver
    driver.expression = expressao_python
    driver.type = 'SCRIPTED'
```

```python
def criar_geometry_nodes(objeto, nome="FGS_GeoNodes"):
    """Cria um modificador de Geometry Nodes programaticamente."""
    mod = objeto.modifiers.new(name=nome, type='NODES')
    node_group = bpy.data.node_groups.new(nome, 'GeometryNodeTree')
    mod.node_group = node_group
    
    nodes = node_group.nodes
    links = node_group.links
    
    # Input e Output padrão
    input_node = nodes.new('NodeGroupInput')
    output_node = nodes.new('NodeGroupOutput')
    
    return node_group, nodes, links
```

### Sistema de Partículas via Python
```python
def criar_sistema_particulas(
    objeto, 
    count=1000, 
    lifetime=60, 
    velocity=(0, 0, 5),
    size=0.05
):
    """
    Cria sistema de partículas completo.
    
    Ideal para: splash, faíscas, glitter, poeira, chuva, neve.
    """
    bpy.context.view_layer.objects.active = objeto
    bpy.ops.object.particle_system_add()
    
    ps = objeto.particle_systems[-1]
    pset = ps.settings
    
    pset.count = count
    pset.lifetime = lifetime
    pset.particle_size = size
    pset.normal_factor = velocity[2]
    pset.factor_random = 0.3
    pset.use_dynamic_rotation = True
    pset.angular_velocity_mode = 'RAND'
    
    return ps
```

```python
def importar_mocap_bvh(caminho_arquivo, escala=0.01):
```

### Compositing em Python (Pós-Produção)
```python
def setup_compositing_cinematografico():
    """
    Ativa compositing com efeitos cinematográficos:
    - Glare (brilho nas luzes)
    - Vignette (escurecimento nas bordas)
    - Color Grading (gradação de cor)
    - Film Grain (grão de filme)
    """
    scene = bpy.context.scene
    scene.use_nodes = True
    tree = scene.node_tree
    nodes = tree.nodes
    links = tree.links
    
    nodes.clear()
    
    # Render Layer
    render_layer = nodes.new('CompositorNodeRLayers')
    
    # Glare (brilho especular)
    glare = nodes.new('CompositorNodeGlare')
    glare.glare_type = 'FOG_GLOW'
    glare.threshold = 0.8
    glare.size = 6
    
    # Lens Distortion (imperfeição óptica realista)
    lens = nodes.new('CompositorNodeLensdist')
    lens.inputs['Distortion'].default_value = -0.05
    lens.inputs['Dispersion'].default_value = 0.02
    
    # Color Balance (grading)
    color_balance = nodes.new('CompositorNodeColorBalance')
    color_balance.correction_method = 'LIFT_GAMMA_GAIN'
    
    # Vignette (bordas escurecidas)
```
